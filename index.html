worker_pool.go
package main

import "sync"

// WorkerPool processes jobs concurrently
func WorkerPool(jobs <-chan Job, n int) {
  var wg sync.WaitGroup

  for i := 0; i < n; i++ {
    wg.Add(1)
    go func() {
      defer wg.Done()
      for j := range jobs {
        j.Process()
      }
    }()
  }
  wg.Wait()
}
