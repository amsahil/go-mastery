Go Cheatsheet

Every essential Go pattern in one dense reference. Copy, paste, ship.

VARIABLES & TYPES
var x int = 10
y := 20                   // short decl
const Pi = 3.14
var s string              // zero value: ""
var p *int               // nil pointer
a, b := 1, 2            // multi-assign
_, err := doThing()     // blank identifier
CONTROL FLOW
// if with init statement
if v, err := parse(s); err != nil {
    log.Fatal(err)
}

// switch (no fallthrough by default)
switch x {
case 1:  fmt.Println("one")
default: fmt.Println("other")
}

// for is the only loop
for i := 0; i < 10; i++ { }
for i, v := range slice { }
for k, v := range myMap { }
SLICES
s := []int{1, 2, 3}
s = append(s, 4, 5)
s2 := make([]int, 0, 10)  // len=0, cap=10
sub := s[1:3]               // [2, 3]
copy(dst, src)
len(s), cap(s)

// 2D slice
grid := make([][]int, rows)
for i := range grid {
    grid[i] = make([]int, cols)
}
MAPS
m := make(map[string]int)
m["key"] = 42
delete(m, "key")

// Check existence
v, ok := m["key"]
if !ok { /* key missing */ }

// Map literal
counts := map[string]int{
    "go":     1,
    "python": 2,
}

// Iterate
for k, v := range m { }
STRUCTS & METHODS
type User struct {
    Name string
    Age  int
}

// Value receiver (no mutation)
func (u User) String() string {
    return u.Name
}

// Pointer receiver (can mutate)
func (u *User) Birthday() {
    u.Age++
}

u := User{Name: "Alice", Age: 30}
u.Birthday()
fmt.Println(u.String())
INTERFACES
type Writer interface {
    Write([]byte) (int, error)
}

// Type assertion
var i interface{} = "hello"
s, ok := i.(string)  // safe

// Type switch
switch v := i.(type) {
case string:  fmt.Println("str", v)
case int:     fmt.Println("int", v)
default:      fmt.Println("other")
}
GOROUTINES & CHANNELS
go doWork()              // launch goroutine

ch := make(chan int)       // unbuffered
ch := make(chan int, 5)    // buffered
ch <- 42                  // send
v := <-ch                  // receive
close(ch)                  // close

var wg sync.WaitGroup
wg.Add(1)
go func() {
    defer wg.Done()
    doWork()
}()
wg.Wait()
ERROR HANDLING
result, err := doThing()
if err != nil {
    return fmt.Errorf("context: %w", err)
}

// Sentinel error
var ErrNotFound = errors.New("not found")

// Check error type
errors.Is(err, ErrNotFound)

// Extract error value
var e *MyError
errors.As(err, &e)

// Panic recovery
defer func() {
    if r := recover(); r != nil { }
}()
NET/HTTP SERVER
http.HandleFunc("/", func(
    w http.ResponseWriter,

    json.NewEncoder(w).Encode(data)

http.ListenAndServe(":8080", nil)

// Custom mux
mux := http.NewServeMux()
mux.HandleFunc("GET /users/{id}", handler)
http.ListenAndServe(":8080", mux)
FILE I/O
// Read entire file
data, err := os.ReadFile("file.txt")

// Write file
os.WriteFile("out.txt", data, 0644)

// Open with defer close
f, err := os.Open("file.txt")
if err != nil { log.Fatal(err) }
defer f.Close()

scanner := bufio.NewScanner(f)
for scanner.Scan() {
    fmt.Println(scanner.Text())
}

GO TOOLS
# Format code
gofmt -w .
goimports -w .

# Lint
go vet ./...
staticcheck ./...

# Dependencies
go mod init github.com/user/pkg
go mod tidy
go get github.com/pkg/errors@v1.2.0

# Build & run
go build -o bin/app .
go run main.go

# Cross compile
GOOS=linux GOARCH=amd64 go build .
